package main

import (
	"fmt"
	"net/http"
	"io/ioutil"
)

func main() {

	url := "http://ec2-3-222-208-106.compute-1.amazonaws.com:8086/orders/{orderId}"

	req, _ := http.NewRequest("PARAMETERS", url, nil)

	res, _ := http.DefaultClient.Do(req)

	defer res.Body.Close()
	body, _ := ioutil.ReadAll(res.Body)

	fmt.Println(res)
	fmt.Println(string(body))

}
